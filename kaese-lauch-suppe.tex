\recipe[2 Portionen]{Käse-Lauch-Suppe}{
	\begin{category}
		\ingredient{2}{Stangen}{Lauch/Porree}
		\ingredient{2}{}{Kartoffeln}
		\ingredient{1}{Zehe}{Knoblauch}
		\ingredient{100}{g}{Tartar}
		\ingredient{100}{ml}{Weißwein, trocken}
		\ingredient{500}{ml}{Gemüsebrühe}
		\ingredient{100}{g}{Schmelzkäse}
		\ingredient{}{}{Salz, Pfeffer, \\ Muskatnuss, Thymian}	
	\end{category}
} {
	\step{
		Lauch in Ringe schneiden, Kartoffeln würfeln, Knoblauch pressen. Öl in einem beschichteten Topf erhitzen, Tatar darin krümelig braten, Gemüse hinzufügen und ca. 5 Minuten anbraten.
	}
	\step{
		Mit Weißwein und Brühe ablöschen, mit Salz und Pfeffer würzen und ca. 15 Minuten köcheln lassen. Schmelzkäse einrühren und mit Muskatnuss und Thymian abschmecken.
	}
}
