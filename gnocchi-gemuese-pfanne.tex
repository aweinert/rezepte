\recipe[3 Portionen]{Gemüsepfanne mit Quark und Gnocchi} {
	\begin{category}
		\ingredient{500}{g}{Gnocchi}
		\ingredient{300}{g}{Buttergemüse\\ (tiefgekühlt)}
		\ingredient{200}{g}{Kräuterquark}
		\ingredient{}{}{Paprikagewürz, Petersilie}
		\ingredient{}{}{Salz, Pfeffer}
	\end{category}
} {
	\step{
		Gemüse auftauen lassen.
		Gnocchi anbraten bis sie knusprig sind.
	}
	\step{
		Mit Quark ablöschen und mit Salz, Pfeffer, Petersilie und Paprika abschmecken.
	}
	\step{
		Gemüse hinzugeben und warm werden lassen.
	}
}
