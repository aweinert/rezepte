\recipe{Gyros-Suppe}{
	\begin{category}
		\ingredient{3}{}{Paprika, bunt}
		\ingredient{3}{}{Gemüsezwiebeln}	
		\ingredient{2}{Zehen}{Knoblauch}
		\ingredient{1}{kg}{Schweineschnitzel}
		\ingredient{500}{g}{Champignons, \\ frisch, klein}
		\ingredient{2}{Konserven}{geschälte \\ Tomaten}
		\ingredient{500}{ml}{Gemüsebrühe}
		\ingredient{160}{g}{Chilisauce}
		\ingredient{200}{g}{Schafskäse/Feta}
		\ingredient{}{}{Salz, Pfeffer, \\ Gyrosgewürz, Oregano}
	\end{category}
} {
	\step[Gyros]{
		Knoblauch fein würfeln.
		Öl in einem großen Topf erhitzen.
		Schweineschnitzeln in feine Streifen schneiden und mit Zwiebeln und Knoblauch darin kräftig anbraten und mit Gyrosgewürz würzen.
	}
	\step[Beilagen]{
		Paprika in Würfel und Zwiebeln in Streifen schneiden.
		Pilze und Paprika kurz mitbraten.
	}
	\step[Soße] {
		Tomaten samt Saft, Gemüsebrühe und Chilisauce zugeben und aufkochen.
		Mit Oregano, Salz und Pfeffer würzen.
	}
	\step{
		Zugedeckt ca. 10 Minuten köcheln.
		Schafskäse zerbröseln, darübergeben und servieren.
	}
}
