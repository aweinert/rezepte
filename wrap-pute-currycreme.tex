\recipe[2 Burritos]{Wrap mit Pute und Currycreme}{
	\begin{category}[Füllung]
		\ingredient{200}{g}{Pu\-ten\-brust\-schni\-tzel}	
		\ingredient{1}{}{Karotte}
		\ingredient{2}{EL}{Mais}
	\end{category}
	
	\begin{category}[Currycreme]
		\ingredient{3}{EL}{Magerquark}	
		\ingredient{1}{EL}{Orangensaft}
	\end{category}
	
	\begin{category}
		\ingredient{2}{}{Tortillawraps}	
		\ingredient{}{}{Rapsöl, Salz, Pfeffer}
		\ingredient{}{}{Majoran, Curry, Petersilie}
	\end{category}
} {
	\step[Füllung]{
		Putenschnitzel in Streifen schneiden, Öl in einer Pfanne erhitzen.
		Schnitzelstreifen darin 3-5 Minuten rundherum braten.
		Mit Salz, Pfeffer und Majoran würzen.
		Karotte raspeln und mit Mais, Petersilie und Putenstreifen mischen.
	}
	\step[Currycreme]{
		Quark mit Orangesaft glatt rühren und mit Salz, Pfeffer und Currypulver würzen.
	}
	\step[Tortillas]{
		Tortillawrap erwärmen, mit Currycreme bestreichen, mit der Geflügel-Gemüse-Füllung belegen und aufrollen.
	}
}
